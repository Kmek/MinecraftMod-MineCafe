public VoxelShape makeShape(){
	VoxelShape shape = VoxelShapes.empty();
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.3125, 0, 0.3125, 0.6875, 0.3125, 0.6875), IBooleanFunction.OR);
	shape = VoxelShapes.join(shape, VoxelShapes.box(0.1875, 0.3125, 0.1875, 0.8125, 0.375, 0.8125), IBooleanFunction.OR);

	return shape;
}
